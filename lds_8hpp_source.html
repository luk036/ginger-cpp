124 template <unsigned long Base = 2> constexpr auto vdc(unsigned long cnt) -> double {
125 auto reslt = 0.0;
126 auto denom = 1.0;
127 auto count = cnt;
128 while (count != 0) {
129 const auto remainder = count % Base;
130 count /= Base;
131 denom *= double(Base);
132 reslt += double(remainder) / denom;
133 }
134 return reslt;
135 }

169 double reverse = 1.0;
